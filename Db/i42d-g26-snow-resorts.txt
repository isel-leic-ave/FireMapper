Resort:
* name: String (PK)
* country: String
* mountain_range: String
* location: String
* coordinates: String

Check_in:
* arrival: String
* departure: String
* departure_city: String
* number_people: number
* transport: String
* price: number
* room: String

Clients:
* civil_id: String (PK)
* name: String
* birth_year: String
* nationality: String
* health_number: number
* covid19_id: String
* email: String
* telephone: number

Example:

Resort {
	name: Les Arcs Panorama
	country: France
	mountain_range: Maciço da Vanoise
	location: Bourg-Saint-Maurice
	coordinates: 45° 34' 18" N 6° 48' 28" E
}
