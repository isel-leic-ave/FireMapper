DRIVER
*number			Number		(PK)
*age			Number
*name			String
*wins			Number
*poles			Number
*wdc_titles		Number

CAR
*chassis		String		(PK)
*power			Number
*displacement		Number
*drivetrain		String
*aspiration		String
*driver			Number		(FK DRIVER.number)

MANUFACTURER
*name			String		(PK)
*number_of_cars		Number
*cars			Array		(FK CAR.chassis)

CATEGORY
*abbreviation		String		(PK)
*full_name		String
*manufacturers		Array		(FK MANUFACTURER.name)
*rules			Array


CHAMPIONSHIP
*abbreviation		String		(PK)
*full_name		String
*category		Array		(FK CAREGORY.abbreviation)

Examples: 

CHAMPIONSHIP {
	abbreviation : "WEC",
	full_name : "World Endurance Championship",
	category : ["Endurance Racing"]
}

	rules : ["4 wheels", "Gasoline", "Hybrid", "Closed Cockpit"]
}
